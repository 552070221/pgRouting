90  std::deque<Path> dag(
91  G &graph,
92  int64_t start_vertex,
93  const std::vector< int64_t > &end_vertex,
94  bool only_cost) {
95  // adjust predecessors and distances vectors
96  clear();
97  size_t n_goals = std::numeric_limits<size_t>::max();
98  predecessors.resize(graph.num_vertices());
99  distances.resize(
100  graph.num_vertices(),
101  std::numeric_limits<double>::infinity());
102 
103 
104  // get the graphs source and target
105  if (!graph.has_vertex(start_vertex))
106  return std::deque<Path>();
107  auto v_source(graph.get_V(start_vertex));
108 
109  std::set< V > s_v_targets;
110  for (const auto &vertex : end_vertex) {
111  if (graph.has_vertex(vertex)) {
112  s_v_targets.insert(graph.get_V(vertex));
113  }
114  }
115 
116  std::vector< V > v_targets(s_v_targets.begin(), s_v_targets.end());
117  // perform the algorithm
118  dag_1_to_many(graph, v_source, v_targets, n_goals);
119 
120  std::deque< Path > paths;
121  // get the results // route id are the targets
122  paths = get_paths(graph, v_source, v_targets, only_cost);
123 
124  std::stable_sort(paths.begin(), paths.end(),
125  [](const Path &e1, const Path &e2)->bool {
126  return e1.end_id() < e2.end_id();
127  });
128 
129  return paths;
130  }

153  // preparation for many to many
154  std::deque<Path> dag(
155  G &graph,
156  const std::vector< int64_t > &start_vertex,
157  const std::vector< int64_t > &end_vertex,
158  bool only_cost) {
159  // a call to 1 to many is faster for each of the sources
160  std::deque<Path> paths;
161 
162  for (const auto &start : start_vertex) {
163  auto r_paths = dag(
164  graph,
165  start, end_vertex,
166  only_cost);
167  paths.insert(paths.begin(), r_paths.begin(), r_paths.end());
168  }
169 
170  std::sort(paths.begin(), paths.end(),
171  [](const Path &e1, const Path &e2)->bool {
172  return e1.end_id() < e2.end_id();
173  });
174  std::stable_sort(paths.begin(), paths.end(),
175  [](const Path &e1, const Path &e2)->bool {
176  return e1.start_id() < e2.start_id();
177  });
178  return paths;
179  }

291  class dijkstra_many_goal_visitor : public boost::default_dijkstra_visitor {
292  public:
293  explicit dijkstra_many_goal_visitor(
294  std::vector< V > goals,
295  size_t n_goals) :
296  m_goals(goals.begin(), goals.end()),
297  m_n_goals(n_goals) {}
298  template <class B_G>
299  void examine_vertex(V u, B_G &) {
300  auto s_it = m_goals.find(u);
301  if (s_it == m_goals.end()) return;
302 
303  // found one more goal
304  m_goals.erase(s_it);
305 
306  // all goals found
307  if (m_goals.size() == 0) throw found_goals();
308 
309  // number of requested goals found
310  --m_n_goals;
311  if (m_n_goals == 0) throw found_goals();
312  }
313 
314  private:
315  std::set< V > m_goals;
316  size_t m_n_goals;
317  };
