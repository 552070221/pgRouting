183  void disconnect_edge(int64_t p_from, int64_t p_to) {
184  V g_from;
185  V g_to;
186  pgr_edge_t d_edge;
187  // nothing to do, the vertex doesnt exist
188  if (!get_gVertex(p_from, g_from)) return;
189  if (!get_gVertex(p_to, g_to)) return;
190  EO_i out, out_end;
191  // store the edges that are going to be removed
192  for (boost::tie(out, out_end) = out_edges(g_from, graph);
193  out != out_end; ++out) {
194  if (target(*out, graph) == g_to) {
195  d_edge.id = graph[*out].id;
196  d_edge.source = graph[source(*out, graph)].id;
197  d_edge.target = graph[target(*out, graph)].id;
198  d_edge.cost = graph[*out].cost;
199  d_edge.reverse_cost = -1;
200  removed_edges.push_back(d_edge);
201  }
202  }
203  // the actual removal
204  boost::remove_edge(g_from, g_to, graph);
205  }

300  get_edge_id(V from, V to, float8 &distance) const {
301  E e;
302  EO_i out_i, out_end;
303  V v_source, v_target;
304  float8 minCost = std::numeric_limits<float8>::max();
305  int64_t minEdge = -1;
306  for (boost::tie(out_i, out_end) = boost::out_edges(from, graph);
307  out_i != out_end; ++out_i) {
308  e = *out_i;
309  v_target = target(e, graph);
310  v_source = source(e, graph);
311  if ((from == v_source) && (to == v_target)
312  && (distance == graph[e].cost))
313  return graph[e].id;
314  if ((from == v_source) && (to == v_target)
315  && (minCost > graph[e].cost)) {
316  minCost = graph[e].cost;
317  minEdge = graph[e].id;
318  }
319  }
320  distance = minEdge == -1? 0: minCost;
321  return minEdge;
322  }
