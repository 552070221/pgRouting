122 void loadDistanceMatrix(vrp_cost_element_t *costmatrix, int cost_count, int depotId)
123 {
124  int i;
125  for(i = 0; i < cost_count; i++)
126  {
127  int fromId = costmatrix[i].src_id;
128  int toId = costmatrix[i].dest_id;
129  CostPack cpack;
130  cpack.cost = costmatrix[i].cost;
131  cpack.distance = costmatrix[i].distance;
132  cpack.traveltime = costmatrix[i].traveltime;
133 
134  if(fromId == depotId)
135  solver.addDepotToOrderCost(fromId, toId, cpack);
136  else if(toId == depotId)
137  solver.addOrderToDepotCost(fromId, toId, cpack);
138  else
139  solver.addOrderToOrderCost(fromId, toId, cpack);
140  }

145 int find_vrp_solution(vrp_vehicles_t *vehicles, int vehicle_count,
146  vrp_orders_t *orders, int order_count,
147  vrp_cost_element_t *costmatrix, int cost_count,
148  int depot_id,
149  vrp_result_element_t **results, int *result_count, char **err_msg)
150 {
151  int res;
152 
153  std::string strError;
154  try {
155  PGR_LOG("Before load order");
156  loadOrders(orders, order_count, depot_id);
157  PGR_LOG("After load order");
158  loadVehicles(vehicles, vehicle_count);
159  PGR_LOG("After load vehicles");
160  loadDistanceMatrix(costmatrix, cost_count, depot_id);
161  PGR_LOG("After load distance matrix");
162  res = solver.solveVRP(strError);
163  PGR_LOG("After VRP Solve");
164 
165  }
166  catch(std::exception& e) {
167  *err_msg = (char *) e.what();
168  return -1;
169  }
170  catch(...) {
171  *err_msg = (char *) "Caught unknown exception!";
172  return -1;
173  }
174 
175 
176  if (res < 0)
177  return res;
178  else
179  {
180  try {
181  CSolutionInfo solution;
182  CTourInfo ctour;
183  // bool bOK =
184  solver.getSolution(solution, strError);
185  int totalRoute = solution.getTourInfoVector().size();
186  int totRows = 0;
187  int i;
188  for(i = 0; i < totalRoute; i++)
189  {
190  totRows += (solution.getTour(i).getServedOrderCount() + 2);
191  }
192  *results = (vrp_result_element_t *) malloc(totRows * sizeof(vrp_result_element_t));
193  *result_count = totRows;
194  int cnt = 0;
195  for(int i = 0; i < totalRoute; i++)
196  {
197  ctour = solution.getTour(i);
198  std::vector<int> vecOrder = ctour.getOrderVector();
199  int totalOrder = vecOrder.size();
200 
201  // For start depot
202  (*results)[cnt].order_id = ctour.getStartDepot();
203  (*results)[cnt].order_pos = 0;
204  (*results)[cnt].vehicle_id = ctour.getVehicleId();
205  (*results)[cnt].arrival_time = -1;
206  (*results)[cnt].depart_time = ctour.getStartTime(0);
207  cnt++;
208 
209  // For each order
210  for(int j = 0; j < totalOrder; j++)
211  {
212  (*results)[cnt].order_id = vecOrder[j];
213  (*results)[cnt].order_pos = j + 1;
214  (*results)[cnt].vehicle_id = ctour.getVehicleId();
215  (*results)[cnt].depart_time = ctour.getStartTime(j + 1);
216  (*results)[cnt].arrival_time = ctour.getStartTime(j + 1) - solver.getServiceTime(vecOrder[j]);
217  cnt++;
218  }
219 
220  // For return depot
221  (*results)[cnt].order_id = ctour.getEndDepot();
222  (*results)[cnt].order_pos = totalOrder + 1;
223  (*results)[cnt].vehicle_id = ctour.getVehicleId();
224  (*results)[cnt].arrival_time = ctour.getStartTime(totalOrder + 1);
225  (*results)[cnt].depart_time = -1;
226  cnt++;
227  }
228  }
229  catch(std::exception& e) {
230  *err_msg = (char *) e.what();
231  return -1;
232  }
233  catch(...) {
234  *err_msg = (char *) "Caught unknown exception!";
235  return -1;
236  }
237  }
238  return EXIT_SUCCESS;
239 }
