159 void BiDirAStar::fconstruct_path(int node_id)
160 {
161  if(m_pFParent[node_id].par_Node == -1)
162  return;
163  fconstruct_path(m_pFParent[node_id].par_Node);
164  path_element_t pt;
165  pt.vertex_id = m_pFParent[node_id].par_Node;
166  pt.edge_id = m_pFParent[node_id].par_Edge;
167  pt.cost = m_pFCost[node_id] - m_pFCost[m_pFParent[node_id].par_Node];
168  m_vecPath.push_back(pt);
169 }

171 /*
172  Reconstruct path for the reverse search. In this case the subsequent node is stored in the parent and the target contains a -1. So one need to add the node
173  and edge to the list and then recurse through the parent upto hitting a -1.
174 */

176 void BiDirAStar::rconstruct_path(int node_id)
177 {
178  path_element_t pt;
179  if(m_pRParent[node_id].par_Node == -1)
180  {
181  pt.vertex_id = node_id;
182  pt.edge_id = -1;
183  pt.cost = 0.0;
184  return;
185  }
186  pt.vertex_id = node_id;
187  pt.cost = m_pRCost[node_id] - m_pRCost[m_pRParent[node_id].par_Node];
188  pt.edge_id = m_pRParent[node_id].par_Edge;
189  m_vecPath.push_back(pt);
190  rconstruct_path(m_pRParent[node_id].par_Node);
191 }

199 void BiDirAStar::explore(int cur_node, double cur_cost, int dir, MinHeap &que)
200 {
201  int i;
202  // Number of connected edges
203  int con_edge = m_vecNodeVector[cur_node].Connected_Edges_Index.size();
204  double edge_cost;
205  for(i = 0; i < con_edge; i++)
206  {
207  int edge_index = m_vecNodeVector[cur_node].Connected_Edges_Index[i];
208  // Get the edge from the edge list.
209  GraphEdgeInfo edge = m_vecEdgeVector[edge_index];
210  // Get the connected node
211  int new_node = m_vecNodeVector[cur_node].Connected_Nodes[i];
212  int mult;
213 
214  if(edge.Direction == 0)
215  mult = 1;
216  else
217  mult = dir;
218  if(cur_node == edge.StartNode)
219  {
220  // Current node is the startnode of the edge. For forward search it should use forward cost, otherwise it should use the reverse cost,
221  // i.e. if the reverse direction is valid then this node may be visited from the end node.
222  if(dir > 0)
223  edge_cost = edge.Cost;
224  else
225  edge_cost = edge.ReverseCost;
226  // Check if the direction is valid for exploration
227  if(edge.Direction == 0 || edge_cost >= 0.0)
228  {
229  //edge_cost = edge.Cost * mult;
230  // Check if the current edge gives better result
231  if(cur_cost + edge_cost < getcost(new_node, dir))
232  {
233  // explore the node, and push it in the queue. the value in the queue will also contain the heuristic cost
234  setcost(new_node, dir, cur_cost + edge_cost);
235  setparent(new_node, dir, cur_node, edge.EdgeID);
236  que.push(std::make_pair(cur_cost + edge_cost + gethcost(new_node, dir), new_node));
237 
238  // Update the minimum cost found so far.
239  if(getcost(new_node, dir) + getcost(new_node, dir * -1) < m_MinCost)
240  {
241  m_MinCost = getcost(new_node, dir) + getcost(new_node, dir * -1);
242  m_MidNode = new_node;
243  }
244  }
245  }
246  }
247  else
248  {
249  // Current node is the endnode of the edge. For forward search it should use reverse cost, otherwise it should use the forward cost,
250  // i.e. if the forward direction is valid then this node may be visited from the start node.
251  if(dir > 0)
252  edge_cost = edge.ReverseCost;
253  else
254  edge_cost = edge.Cost;
255  // Check if the direction is valid for exploration
256  if(edge.Direction == 0 || edge_cost >= 0.0)
257  {
258  //edge_cost = edge.ReverseCost * mult;
259 
260  // Check if the current edge gives better result
261  if(cur_cost + edge_cost < getcost(new_node, dir))
262  {
263  // explore the node, and push it in the queue. the value in the queue will also contain the heuristic cost
264  setcost(new_node, dir, cur_cost + edge_cost);
265  setparent(new_node, dir, cur_node, edge.EdgeID);
266  que.push(std::make_pair(cur_cost + edge_cost + gethcost(new_node, dir), new_node));
267  // Update the minimum cost found so far.
268  if(getcost(new_node, dir) + getcost(new_node, dir * -1) < m_MinCost)
269  {
270  m_MinCost = getcost(new_node, dir) + getcost(new_node, dir * -1);
271  m_MidNode = new_node;
272  }
273  }
274  }
275  }

284 int BiDirAStar:: bidir_astar(edge_astar_t *edges, unsigned int edge_count, int maxNode, int start_vertex, int end_vertex,
285  path_element_t **path, int *path_count, char **err_msg)
286 {
287  max_node_id = maxNode;
288  max_edge_id = -1;
289 
290  // construct the graph from the edge list, i.e. populate node and edge data structures
291  construct_graph(edges, edge_count, maxNode);
292 
293  m_lStartNodeId = start_vertex;
294  m_lEndNodeId = end_vertex;
295 
296  int nodeCount = m_vecNodeVector.size();
297 
298  MinHeap fque(maxNode + 2);
299  MinHeap rque(maxNode + 2);
300  //std::priority_queue<PDI, std::vector<PDI>, std::greater<PDI> > fque;
301  //std::priority_queue<PDI, std::vector<PDI>, std::greater<PDI> > rque;
302 
303  m_vecPath.clear();
304 
305  int i;
306  // Allocate memory for local storage like cost and parent holder
307  initall(maxNode);
308 
309  // Initialize the forward search
310  m_pFParent[start_vertex].par_Node = -1;
311  m_pFParent[start_vertex].par_Edge = -1;
312  m_pFCost[start_vertex] = 0.0;
313  fque.push(std::make_pair(0.0, start_vertex));
314 
315  // Initialize the reverse search
316  m_pRParent[end_vertex].par_Node = -1;
317  m_pRParent[end_vertex].par_Edge = -1;
318  m_pRCost[end_vertex] = 0.0;
319  rque.push(std::make_pair(0.0, end_vertex));
320 
321 
322  int new_node;
323  int cur_node;
324  int dir;
325 /*
326  The main loop. The algorithm is as follows:
327  1. IF the sum of the current minimum of both heap is greater than so far found path, we cannot get any better, so break the loop.
328  2. IF the reverse heap minimum is lower than the forward heap minimum, explore from reverse direction.
329  3. ELSE explore from the forward directtion.
330 */
331 
332  while(!fque.empty() && !rque.empty())
333  {
334  PDI fTop = fque.top();
335  PDI rTop = rque.top();
336  if(m_pFCost[fTop.second] + m_pRCost[rTop.second] > m_MinCost) //We are done, there is no path with lower cost
337  break;
338 
339  if(rTop.first < fTop.first) // Explore from reverse queue
340  {
341  if(rTop.first > m_MinCost)
342  break;
343  cur_node = rTop.second;
344  int dir = -1;
345  rque.pop();
346  explore(cur_node, m_pRCost[rTop.second], dir, rque);
347  }
348  else // Explore from forward queue
349  {
350  if(fTop.first > m_MinCost)
351  break;
352  cur_node = fTop.second;
353  int dir = 1;
354  fque.pop();
355  explore(cur_node, m_pFCost[fTop.second], dir, fque);
356  }
357  }
358 
359 /*
360  Path reconstruction part. m_MidNode is the joining point where two searches meet to make a shortest path. It is updated in explore.
361  If it contains -1, then no path is found. Other wise we have a shortest path and that is reconstructed in the m_vecPath.
362 */
363 
364  if(m_MidNode == -1)
365  {
366  *err_msg = (char *)"Path Not Found";
367  deleteall();
368  return -1;
369  }
370  else
371  {
372  // reconstruct path from forward search
373  fconstruct_path(m_MidNode);
374  // reconstruct path from backward search
375  rconstruct_path(m_MidNode);
376 
377  // insert the last row in the path trace (having edge_id = -1 and cost = 0.0)
378  path_element_t pelement;
379  pelement.vertex_id = end_vertex;
380  pelement.edge_id = -1;
381  pelement.cost = 0.0;
382  m_vecPath.push_back(pelement);
383 
384  // Transfer data path to path_element_t format and allocate memory and populate the pointer
385  *path = (path_element_t *) malloc(sizeof(path_element_t) * (m_vecPath.size() + 1));
386  *path_count = m_vecPath.size();
387 
388  for(i = 0; i < *path_count; i++)
389  {
390  (*path)[i].vertex_id = m_vecPath[i].vertex_id;
391  (*path)[i].edge_id = m_vecPath[i].edge_id;
392  (*path)[i].cost = m_vecPath[i].cost;
393  }
394 
395  }
396  deleteall();
397  return 0;
398 }

435 bool BiDirAStar::addEdge(edge_astar_t edgeIn)
436 {
437  long lTest;
438  // Check if the edge is already processed.
439  Long2LongMap::iterator itMap = m_mapEdgeId2Index.find(edgeIn.id);
440  if(itMap != m_mapEdgeId2Index.end())
441  return false;
442 
443  // Create a GraphEdgeInfo using the information of the current edge
444  GraphEdgeInfo newEdge;
445  newEdge.EdgeID = edgeIn.id;
446  newEdge.EdgeIndex = m_vecEdgeVector.size();
447  newEdge.StartNode = edgeIn.source;
448  newEdge.EndNode = edgeIn.target;
449  newEdge.Cost = edgeIn.cost;
450  newEdge.ReverseCost = edgeIn.reverse_cost;
451 
452  // Set the direction. If both cost and reverse cost has positive value the edge is bidirectional and direction field is 0. If cost is positive and reverse cost
453  // negative then the edge is unidirectional with direction = 1 (goes from source to target) otherwise it is unidirectional with direction = -1 (goes from target
454  // to source). Another way of creating unidirectional edge is assigning big values in cost or reverse_cost. In that case the direction is still zero and this case
455  // is handled in the algorithm automatically.
456  if(newEdge.Cost >= 0.0 && newEdge.ReverseCost >= 0)
457  {
458  newEdge.Direction = 0;
459  }
460  else if(newEdge.Cost >= 0.0)
461  {
462  newEdge.Direction = 1;
463  }
464  else
465  {
466  newEdge.Direction = -1;
467  }
468 
469  if(edgeIn.id > max_edge_id)
470  {
471  max_edge_id = edgeIn.id;
472  }
473 
474  // Update max_edge_id

485  m_vecNodeVector[newEdge.StartNode].ypos = edgeIn.s_y;
486
