105  tmp.edge = -1;
106  tmp.cost = 0;
107  tmp.agg_cost = totalcost;
108  tmp.tree_cost = totalcost;

110  results.push_back(tmp);

112 
113  for (size_t j = 1; j < size; j++){
114  pgr_prim_t tmp;
115 
116  tmp.root_vertex = root_vertex; // root_vertex
117  auto start_node = graph.graph[predecessors[data[j]]].id;
118  tmp.node = graph.graph[data[j]].id; // node
119 
120  auto v_sn(graph.get_V(start_node));
121  auto v_en(graph.get_V(tmp.node));
122 
123  auto cost = distances[v_sn] - distances[v_en];
124  auto edge_id =
125  graph.get_edge_id(v_sn, v_en, cost);
126  totalcost += cost;
127 
128  tmp.edge = edge_id; // edge_id
129  tmp.cost = cost; // cost
130  tmp.agg_cost = aggegrateCost(v_root, data[j]); // agg_cost
131  tmp.tree_cost = totalcost; // tree_cost
132  results.push_back(tmp);
133  }
