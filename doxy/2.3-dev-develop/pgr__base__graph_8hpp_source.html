593 void
594 Pgr_base_graph< G, T_V, T_E >::disconnect_edge(int64_t p_from, int64_t p_to) {
595  T_E d_edge;
596 
597  // nothing to do, the vertex doesnt exist
598  if (!has_vertex(p_from) || !has_vertex(p_to)) return;
599 
600  EO_i out, out_end;
601  V g_from(get_V(p_from));
602  V g_to(get_V(p_to));
603 
604  // store the edges that are going to be removed
605  for (boost::tie(out, out_end) = out_edges(g_from, graph);
606  out != out_end; ++out) {
607  if (target(*out, graph) == g_to) {
608  d_edge.id = graph[*out].id;
609  d_edge.source = graph[source(*out, graph)].id;
610  d_edge.target = graph[target(*out, graph)].id;
611  d_edge.cost = graph[*out].cost;
612  removed_edges.push_back(d_edge);
613  }
614  }
615  // the actual removal
616  boost::remove_edge(g_from, g_to, graph);
617 }

622 void
623 Pgr_base_graph< G, T_V, T_E >::disconnect_out_going_edge(int64_t vertex_id, int64_t edge_id) {
624  T_E d_edge;
625 
626  // nothing to do, the vertex doesnt exist
627  if (!has_vertex(vertex_id)) return;
628  auto v_from(get_V(vertex_id));
629 
630  EO_i out, out_end;
631  bool change = true;
632  // store the edge that are going to be removed
633  while (change) {
634  change = false;
635  for (boost::tie(out, out_end) = out_edges(v_from, graph);
636  out != out_end; ++out) {
637  if (graph[*out].id == edge_id) {
638  d_edge.id = graph[*out].id;
639  d_edge.source = graph[source(*out, graph)].id;
640  d_edge.target = graph[target(*out, graph)].id;
641  d_edge.cost = graph[*out].cost;
642  removed_edges.push_back(d_edge);
643  boost::remove_edge((*out), graph);
644  change = true;
645  break;
646  }
647  }
648  }

702 template < class G, typename T_V, typename T_E >
703 int64_t
704 Pgr_base_graph< G, T_V, T_E >::get_edge_id(V from, V to, double &distance) const {
705  E e;
706  EO_i out_i, out_end;
707  V v_source, v_target;
708  double minCost = std::numeric_limits<double>::max();
709  int64_t minEdge = -1;
710  for (boost::tie(out_i, out_end) = boost::out_edges(from, graph);
711  out_i != out_end; ++out_i) {
712  e = *out_i;
713  v_target = target(e, graph);
714  v_source = source(e, graph);
715  if ((from == v_source) && (to == v_target)
716  && (distance == graph[e].cost))
717  return graph[e].id;
718  if ((from == v_source) && (to == v_target)
719  && (minCost > graph[e].cost)) {
720  minCost = graph[e].cost;
721  minEdge = graph[e].id;
722  }
723  }
724  distance = minEdge == -1? 0: minCost;
725  return minEdge;
726 }

731 Pgr_base_graph< G, T_V, T_E >::graph_add_edge(const T_E &edge ) {
732  bool inserted;

734  typename Pgr_base_graph< G, T_V, T_E >::E e;

736  vm_s = vertices_map.find(edge.source);
737  if (vm_s == vertices_map.end()) {
738  vertices_map[edge.source]= m_num_vertices;
739  vm_s = vertices_map.find(edge.source);
740  }
741 
742  vm_t = vertices_map.find(edge.target);
743  if (vm_t == vertices_map.end()) {
744  vertices_map[edge.target]= m_num_vertices;
745  vm_t = vertices_map.find(edge.target);
746  }
747 
748  if (edge.cost >= 0) {
749  boost::tie(e, inserted) =
750  boost::add_edge(vm_s->second, vm_t->second, graph);
751  graph[e].cp_members(edge);
752  }
753 }

756 template < class G, typename T_V, typename T_E >
757 template < typename T>
758 void
759 Pgr_base_graph< G, T_V, T_E >::graph_add_edge(const T &edge) {
760  bool inserted;
761  typename Pgr_base_graph< G, T_V, T_E >::E e;
762  if ((edge.cost < 0) && (edge.reverse_cost < 0))
763  return;
764 
765  /*
766  * true: for source
767  * false: for target
768  */
769  auto vm_s = get_V(T_V(edge, true));
770  auto vm_t = get_V(T_V(edge, false));
771 
772  pgassert(vertices_map.find(edge.source) != vertices_map.end());
773  pgassert(vertices_map.find(edge.target) != vertices_map.end());
774  if (edge.cost >= 0) {
775  boost::tie(e, inserted) =
776  boost::add_edge(vm_s, vm_t, graph);
777  graph[e].cost = edge.cost;
778  graph[e].id = edge.id;
779  graph[e].first = true;
780  }
781 
782  if (edge.reverse_cost >= 0) {
783  boost::tie(e, inserted) =
784  boost::add_edge(vm_t, vm_s, graph);
785 
786  graph[e].cost = edge.reverse_cost;
787  graph[e].id = edge.id;
788  graph[e].first = false;
789  }
790 }
