151  // euclidean distance heuristic for one goal
152  class distance_heuristic : public boost::astar_heuristic< B_G, double > {
153  public:
154  distance_heuristic(B_G &g, V goal, int heuristic, double factor)
155  : m_g(g),
156  m_factor(factor),
157  m_heuristic(heuristic) {
158  m_goals.insert(goal);
159  }
160  distance_heuristic(B_G &g, std::vector< V > goals, int heuristic, double factor)
161  : m_g(g),
162  m_goals(goals.begin(), goals.end()),
163  m_factor(factor),
164  m_heuristic(heuristic) {}
165 
166  double operator()(V u) {
167  if (m_heuristic == 0) return 0;
168  if (m_goals.empty()) return 0;
169  double best_h(std::numeric_limits<double>::max());
170  for (auto goal : m_goals) {
171  double current(std::numeric_limits<double>::max());
172  double dx = m_g[goal].x() - m_g[u].x();
173  double dy = m_g[goal].y() - m_g[u].y();
174  switch (m_heuristic) {
175  case 0:
176  current = 0;
177  case 1:
178  current = std::fabs(std::max(dx, dy)) * m_factor;
179  case 2:
180  current = std::fabs(std::min(dx, dy)) * m_factor;
181  case 3:
182  current = (dx * dx + dy * dy) * m_factor * m_factor;
183  case 4:
184  current = std::sqrt(dx * dx + dy * dy) * m_factor;
185  case 5:
186  current = (std::fabs(dx) + std::fabs(dy)) * m_factor;
187  default:
188  current = 0;
189  }
190  if (current < best_h) {
191  best_h = current;
192  };
193  }
194  {
195  auto s_it = m_goals.find(u);
196  if (!(s_it == m_goals.end())) {
197  // found one more goal
198  m_goals.erase(s_it);
199  }
200  }
201  return best_h;
202  };
203 
204  private:
205  B_G &m_g;
206  std::set< V > m_goals;
207  double m_factor;
208  int m_heuristic;
209  }; // class distance_heuristic

493 template < class G >
494 void
495 Pgr_astar< G >::get_path(
496  const G &graph,
497  V source,
498  V target,
499  Path &r_path) const {
500  // backup of the target
501  V target_back = target;
502  uint64_t from(graph.graph[source].id);
503  uint64_t to(graph.graph[target].id);
504 
505  // no path was found
506  if (target == predecessors[target]) {
507  r_path.clear();
508  return;
509  }
510 
511  // findout how large is the path
512  int64_t result_size = 1;
513  while (target != source) {
514  if (target == predecessors[target]) break;
515  result_size++;
516  target = predecessors[target];
517  }
518 
519  // recover the target
520  target = target_back;
521 
522  // variables that are going to be stored
523  int64_t vertex_id;
524  int64_t edge_id;
525  double cost;
526 
527  // working from the last to the beginning
528 
529  // initialize the sequence
530  auto seq = result_size;
531  // the last stop is the target
532  Path path(from, to);
533  path.push_front(
534  {graph.graph[target].id, -1,
535  0, distances[target]});
536 
537  while (target != source) {
538  // we are done when the predecesor of the target is the target
539  if (target == predecessors[target]) break;
540  // values to be inserted in the path
541  --seq;
542  cost = distances[target] - distances[predecessors[target]];
543  vertex_id = graph.graph[predecessors[target]].id;
544  edge_id = graph.get_edge_id(predecessors[target], target, cost);
545 
546  path.push_front({vertex_id, edge_id,
547  cost, (distances[target] - cost)});
548  target = predecessors[target];
549  }
550  r_path = path;
551  return;
552 }
