134  /*
135  * sort the paths by size from greater to smaller
136  * and sort each path by node
137  * all the nodes on p2 are going to be compared
138  * with the nodes of p1
139  *
140  * When both paths reach the node and p1.agg_cost > p2.agg_cost
141  * erase the node of p1
142  * (cant erase from p2 because we loose the iterators
143  * so in a future cycle it will be deleted)
144  *
145  * sort the paths by start_id,
146  */
147 
148  friend void equi_cost(std::deque< Path > &paths) {
149 
150  /* sort paths by size: largest first */
151  std::sort(paths.begin(), paths.end(),
152  [](const Path &e1, const Path &e2)->bool {
153  return e2.size() < e1.size();
154  });
155 
156  /* sort each path by node: smaller id first */
157  for (auto &p : paths) {
158  if (p.size() < 2) continue;
159  std::sort(p.begin(), p.end(),
160  [](const Path_t &e1, const Path_t &e2)->bool {
161  return e1.node < e2.node;
162  });
163  }
164 
165  for (auto &p1 : paths) {
166  for (const auto &p2 : paths) {
167  if (p1.start_id() == p2.start_id()) continue;
168  for (const auto &stop : p2.path) {
169  /* find the node of p2 in p1 */
170  auto pos = lower_bound(p1.begin(), p1.end(), stop,
171  [](const Path_t &l, const Path_t &r )->bool {
172  return l.node < r.node;
173  });
174 
175  if (pos != p1.end() && stop.node == pos->node && stop.agg_cost < pos->agg_cost) {
176  /* both share the same node &
177  * the second path has the smallest
178  * So erasing from the first path */
179  p1.erase(pos);
180  }
181  }
182  }
183  }
184 
185  /* sort paths by start_id */
186  std::sort(paths.begin(), paths.end(),
187  [](const Path &e1, const Path &e2)->bool {
188  return e1.start_id() < e2.start_id();
189  });
190 
191  /* sort each path by agg_cost, node */
192  for (auto &path : paths) {
193  /* least influential data first */
194  std::sort(path.begin(), path.end(),
195  [](const Path_t &l, const Path_t &r)
196  { return l.node < r.node;});
197  /* preserve the order of what we did before */
198  std::stable_sort(path.begin(), path.end(),
199  [](const Path_t &l, const Path_t &r)
200  { return l.agg_cost < r.agg_cost;});
201  }
202  }
