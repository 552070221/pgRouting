158 Pgr_lineGraph< G, T_V, T_E >::get_ids(std::vector< T >& restrictions) {
159  for (auto &r: restrictions) {
160  auto restrict_edges = r.restrict_edges();
161  std::vector < int64_t > temp;
162 
163  pgassert(m_edges.find(restrict_edges[0]) != m_edges.end());
164  auto prev = m_edges[restrict_edges[0]];
165 
166  for (auto i = 1; i < (int64_t)restrict_edges.size(); i++) {
167  pgassert(m_edges.find(restrict_edges[i]) != m_edges.end());
168  auto cur = m_edges[restrict_edges[i]];
169 
170  if (prev.target == cur.target) {
171  std::swap(cur.source, cur.target);
172  std::swap(cur.cost, cur.reverse_cost);
173  }
174 
175  if(prev.source == cur.source) {
176  std::swap(prev.source, prev.target);
177  std::swap(prev.cost, prev.reverse_cost);
178  }
179 
180  if(prev.source == cur.target) {
181  std::swap(prev.source, prev.target);
182  std::swap(prev.cost, prev.reverse_cost);
183  std::swap(cur.source, cur.target);
184  std::swap(cur.cost, cur.reverse_cost);
185  }
186 
187  pgassert(m_vertex_map.find( {prev.id, prev.source} ) != m_vertex_map.end());
188  pgassert(m_vertex_map.find( {cur.id, cur.source} ) != m_vertex_map.end());
189 
190  if (temp.empty()) {
191  temp.push_back( m_vertex_map[ {prev.id, prev.source} ] );
192  }
193
