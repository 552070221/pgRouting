78 double
79 eucledianDmatrix::tourCost(const Tour &tour) const {
80  double total_cost(0);
81  if (tour.cities.empty()) return total_cost;
82 
83  auto prev_id = tour.cities.front();
84  for (const auto &id : tour.cities) {
85  if (id == tour.cities.front()) continue;
86 
87  total_cost += distance(prev_id, id);
88  prev_id = id;
89  }
90  total_cost += distance(prev_id, tour.cities.front());
91  return total_cost;
92 }
