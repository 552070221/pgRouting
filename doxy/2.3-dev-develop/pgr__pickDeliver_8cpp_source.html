131 Pgr_pickDeliver::Pgr_pickDeliver(
132  const Customer_t *customers_data, size_t total_customers,
133  int p_max_vehicles,
134  double p_capacity,
135  double p_speed,
136  size_t p_max_cycles,
137  std::string &error) :
138  /* Set the depot to be the first ddata found */
139  max_capacity(p_capacity),
140  m_speed(p_speed),
141  m_max_cycles(p_max_cycles),
142  max_vehicles(p_max_vehicles),
143  m_starting_site({0, customers_data[0], Tw_node::NodeType::kStart, this}),
144  m_ending_site({0, customers_data[0], Tw_node::NodeType::kEnd, this}),
145  m_original_data(customers_data, customers_data + total_customers) {
146  pgassert(m_speed > 0);
147  pgassert(m_max_cycles > 0);
148  pgassert(max_vehicles > 0);
149  std::ostringstream tmplog;
150  error = "";
151 
152  log << "\n *** Constructor of problem ***\n";
153 
154  /* sort data by id */
155  std::sort(m_original_data.begin(), m_original_data.end(),
156  [] (const Customer_t &c1, const Customer_t &c2)
157  {return c1.id < c2.id;});
158 
159  /*
160  * starting node:
161  * id must be 0
162  */
163  if (m_original_data[0].id != 0) {
164  error = "Depot node not found";
165  return;
166  }
167 
168  m_starting_site = Vehicle_node(
169  {0, customers_data[0], Tw_node::NodeType::kStart, this});
170  m_ending_site = Vehicle_node(
171  {1, customers_data[0], Tw_node::NodeType::kEnd, this});
172  if (!m_starting_site.is_start()) {
173  log << "DEPOT" << m_starting_site;
174  error = "Illegal values found on the starting site";
175  return;
176  }
177  pgassert(m_starting_site.is_start());
178  pgassert(m_ending_site.is_end());
179 
180  m_nodes.push_back(m_starting_site);
181  m_nodes.push_back(m_ending_site);
182 
183 
184  ID order_id(0);
185  ID node_id(2);
186  for (const auto p : m_original_data) {
187  /*
188  * skip Starting site
189  */
190  if (p.id == 0) continue;
191 
192  /*
193  * SAMPLE CORRECT INFORMATION
194  *
195  * The Pickup is 11 (pindex == 0)
196  * The Deliver is 1 (dindex == 0)
197  *
198  * id | x | y | demand | etime | Ltime | Stime | pindex | dindex
199  * 1 | 45 | 68 | -10 | 912 | 967 | 90 | 11 | 0
200  * 11 | 35 | 69 | 10 | 448 | 505 | 90 | 0 | 1
201  *
202  */
203 
204  /*
205  * skip deliveries
206  */
207  if (p.Dindex == 0) continue;
208 
209  /*
210  * pickup is found
211  */
212  Tw_node pickup({node_id++, p, Tw_node::NodeType::kPickup, this});
213  if (!pickup.is_pickup()) {
214  log << "PICKUP" << pickup;
215  tmplog << "Illegal values found on Pickup " << p.id;
216  error = tmplog.str();
217  return;
218  }
219  pgassert(pickup.is_pickup());
220 
221 
222  /*
223  * look for corresponding the delivery of the pickup
224  */
225  auto deliver_ptr = std::lower_bound(
226  m_original_data.begin(),
227  m_original_data.end(),
228  p,
229  [] (const Customer_t &delivery, const Customer_t &pick)
230  -> bool
231  {return delivery.id < pick.Dindex;});
232 
233  if (deliver_ptr == m_original_data.end()
234  || deliver_ptr->id != p.Dindex) {
235  tmplog << "For Pickup "
236  << p.id
237  << " the corresponding Delivery was not found";
238  error = tmplog.str();
239  return;
240  }
241 
242 
243  /*
244  * delivery is found
245  */
246  Tw_node delivery(
247  node_id++,
248  (*deliver_ptr),
249  Tw_node::NodeType::kDelivery,
250  this);
251  if (!delivery.is_delivery()) {
252  log << "DELIVERY" << delivery;
253  tmplog << "Illegal values found on Delivery "
254  << deliver_ptr->id;
255  error = tmplog.str();
256  return;
257  }
258  pgassert(delivery.is_delivery());
259 
260 
261  /*
262  * add into an order & check the order
263  */
264  pickup.set_Did(delivery.id());
265  delivery.set_Pid(pickup.id());
266  m_nodes.push_back(pickup);
267  m_nodes.push_back(delivery);
268 
269  m_orders.push_back(
270  Order(order_id, node(node_id - 2),
271  node(node_id - 1),
272  this));
273  pgassert(m_orders.back().pickup().is_pickup());
274  pgassert(m_orders.back().delivery().is_delivery());
275  pgassert(static_cast<Tw_node> (m_orders.back().pickup()) == pickup);
276 
277  /*
278  * check the (S, P, D, E) order
279  */
280  {
281  Vehicle_pickDeliver truck(
282  order_id,
283  m_starting_site,
284  m_ending_site,
285  max_capacity,
286  this);
287  truck.push_back(m_orders.back());
288 
289  if (!truck.is_feasable()) {
290  log << truck << "\n";
291  tmplog << "The (pickup, delivery) = ("
292  << m_orders.back().pickup().original_id() << ", "
293  << m_orders.back().delivery().original_id()
294  << ") is not feasible";
295  error = tmplog.str();
296  return;
297  }
298  pgassert(truck.is_feasable());
299  } // check
300 
301  ++order_id;
302  } // for
303 
304  /*
305  * double check we found all orders
306  */
307  if (((m_orders.size() * 2 + 1) - m_original_data.size()) != 0) {
308  error = "A pickup was not found";
309  return;
310  }
311 
312  for (auto &o : m_orders) {
313  o.setCompatibles();
314  }
315 
316  for (auto o : m_orders) {
317  log << o;
318  }
319  } // constructor
