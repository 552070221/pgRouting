74 int64_t* pgr_get_bigIntArray(int *arrlen, ArrayType *input) {
75  int ndims;
76  // int *lbs;
77  bool *nulls;
78  Oid i_eltype;
79  int16 i_typlen;
80  bool i_typbyval;
81  char i_typalign;
82  Datum *i_data;
83  int i, n;
84  int64_t *data;
85 
86  /* get input array element type */
87  i_eltype = ARR_ELEMTYPE(input);
88  get_typlenbyvalalign(i_eltype, &i_typlen, &i_typbyval, &i_typalign);
89 
90 
91  /* validate input data type */
92  switch(i_eltype){
93  case INT2OID:
94  case INT4OID:
95  case INT8OID:
96  break;
97  default:
98  elog(ERROR, "Expected array of any-integer");
99  return (int64_t*) NULL;
100  break;
101  }
102 
103  /* get various pieces of data from the input array */
104  ndims = ARR_NDIM(input);
105  n = (*ARR_DIMS(input));
106  (*arrlen) = n;
107  // lbs = ARR_LBOUND(input);
108 
109  if ( (ndims) != 1) {
110  elog(ERROR, "One dimenton expected");
111  }
112 
113  /* get src data */
114  deconstruct_array(input, i_eltype, i_typlen, i_typbyval, i_typalign,
115  &i_data, &nulls, &n);
116 
117  /* construct a C array */
118  data = (int64_t *) malloc((*arrlen) * sizeof(int64_t));
119  if (!data) {
120  elog(ERROR, "Error: Out of memory!");
121  }
122  PGR_DBG("array size %d", (*arrlen));
123 
124  for (i=0; i<(*arrlen); i++) {
125  if (nulls[i]) {
126  data[i] = -1;
127  }
128  else {
129  switch(i_eltype){
130  case INT2OID:
131  data[i] = (int64_t) DatumGetInt16(i_data[i]);
132  break;
133  case INT4OID:
134  data[i] = (int64_t) DatumGetInt32(i_data[i]);
135  break;
136  case INT8OID:
137  data[i] = DatumGetInt64(i_data[i]);
138  break;
139  }
140  }
141  PGR_DBG(" data[%d]=%li", i, data[i]);
142  }
143 
144  pfree(nulls);
145  pfree(i_data);
146 
147  return (int64_t*)data;
148 }

252 int pgr_get_data(
253  char *sql,
254  pgr_edge_t **edges,
255  int64_t *totalTuples,
256  bool has_rcost,
257  int64_t start_vertex,
258  int64_t end_vertex) {
259  PGR_DBG("Entering pgr_get_data");
260 
261  bool sourceFound = false;
262  bool targetFound = false;
263  if (start_vertex == -1 && end_vertex == -1) {
264  sourceFound = targetFound = true;
265  }
266  int ntuples;
267  int64_t total_tuples;
268 
269  int edge_columns[5];
270  int edge_types[5];
271  int i;
272  for (i = 0; i < 5; ++i) edge_columns[i] = -1;
273  for (i = 0; i < 5; ++i) edge_types[i] = -1;
274  int ret = -1;
275 
276 
277  PGR_DBG("Connecting to SPI");
278  int SPIcode;
279  SPIcode = SPI_connect();
280  if (SPIcode != SPI_OK_CONNECT) {
281  elog(ERROR, "Couldn't open a connection to SPI");
282  return -1;
283  }
284 
285  PGR_DBG("Preparing Plan");
286  void *SPIplan;
287  SPIplan = SPI_prepare(sql, 0, NULL);
288  if (SPIplan == NULL) {
289  elog(ERROR, "Couldn't create query plan via SPI");
290  return -1;
291  }
292 
293  PGR_DBG("Opening Portal");
294  Portal SPIportal;
295  if ((SPIportal = SPI_cursor_open(NULL, SPIplan, NULL, NULL, true)) == NULL) {
296  elog(ERROR, "SPI_cursor_open('%s') returns NULL", sql);
297  return -1;
298  }
299 
300 
301  PGR_DBG("Starting Cycle");
302  bool moredata = TRUE;
303  (*totalTuples) = total_tuples = 0;
304  while (moredata == TRUE) {
305  SPI_cursor_fetch(SPIportal, TRUE, TUPLIMIT);
306 
307  /* on the first tuple get the column numbers */
308  if (edge_columns[0] == -1) {
309  PGR_DBG("Fetching column numbers");
310  if (pgr_fetch_edge_columns(&edge_columns, &edge_types,
311  has_rcost) == -1)
312  return pgr_finish(SPIcode, ret);
313  PGR_DBG("Finished fetching column numbers");
314  }
315 
316  ntuples = SPI_processed;
317  total_tuples += ntuples;
318 
319  PGR_DBG("Getting Memory");
320  if ((*edges) == NULL)
321  (*edges) = (pgr_edge_t *)palloc(total_tuples * sizeof(pgr_edge_t));
322  else
323  (*edges) = (pgr_edge_t *)repalloc((*edges), total_tuples * sizeof(pgr_edge_t));
324  PGR_DBG("Got Memory");
325 
326  if ((*edges) == NULL) {
327  elog(ERROR, "Out of memory");
328  return pgr_finish(SPIcode, ret);
329  }
330 
331  if (ntuples > 0) {
332  int t;
333  SPITupleTable *tuptable = SPI_tuptable;
334  TupleDesc tupdesc = SPI_tuptable->tupdesc;
335  PGR_DBG("processing %d", ntuples);
336  for (t = 0; t < ntuples; t++) {
337  HeapTuple tuple = tuptable->vals[t];
338  pgr_fetch_edge(&tuple, &tupdesc, &edge_columns, &edge_types,
339  &(*edges)[total_tuples - ntuples + t], has_rcost);
340 
341  if (!sourceFound
342  && (((*edges)[total_tuples - ntuples + t].source == start_vertex)
343  || ((*edges)[total_tuples - ntuples + t].target == start_vertex))) {
344  sourceFound = true;
345  }
346  if (!targetFound
347  && (((*edges)[total_tuples - ntuples + t].source == end_vertex)
348  || ((*edges)[total_tuples - ntuples + t].target == end_vertex))) {
349  targetFound = true;
350  }
351  }
352  SPI_freetuptable(tuptable);
353  } else {
354  moredata = FALSE;
355  }
356  }
357 
358 
359 #if 0
360  if (!sourceFound) {
361  // elog(NOTICE, "Starting Vertex does not exist in the data");
362  return 0;
363  }
364  if (!targetFound) {
365  // elog(NOTICE, "Ending Vertex does not exist in the data");
366  return 0;
367  }
368 #endif
369  if (total_tuples == 1) {
370  // for some reason it needs at least a second edge for boost.graph to work
371  // makeing a simple test and asking boost people
372  ++total_tuples;
373  (*edges) = (pgr_edge_t *)repalloc((*edges), total_tuples * sizeof(pgr_edge_t));
374  (*edges)[1].source = -1;
375  (*edges)[1].target = -1;
376  (*edges)[1].cost = 10000;
377  (*edges)[1].id = (*edges)[0].id + 1;
378  (*edges)[1].reverse_cost = -1;
379  }
380 
381  (*totalTuples) = total_tuples;
382  return 0;
383 }
