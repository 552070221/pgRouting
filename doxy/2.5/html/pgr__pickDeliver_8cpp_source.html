194 Pgr_pickDeliver::order_of(const Vehicle_node &node) const {
195  pgassert(node.is_pickup() || node.is_delivery());
196  if (node.is_pickup()) {
197  for (const auto o : m_orders) {
198  if (o.pickup().id() == node.id()) {
199  return o;
200  }
201  }
202  }
203  pgassert(node.is_delivery());
204 
205  for (const auto o : m_orders) {
206  if (o.delivery().id() == node.id()) {
207  return o;

209  }

213 #endif
