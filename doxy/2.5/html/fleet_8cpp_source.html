97 Fleet::build_fleet(
98  std::vector<Vehicle_t> vehicles,
99  size_t &node_id) {
100  /*
101  * creating a phoney truck with max capacity and max window
102  */
103  vehicles.push_back({
104  -1,
105  std::numeric_limits<double>::infinity(),
106  vehicles[0].speed,
107  vehicles[0].start_x,
108  vehicles[0].start_y,
109  1,
110  0,
111  std::numeric_limits<double>::infinity(),
112  0,
113  vehicles[0].end_x,
114  vehicles[0].end_y,
115  0,
116  std::numeric_limits<double>::infinity(),
117  0});
118 
119 
120  for (auto vehicle : vehicles) {
121 
122  if (vehicle.cant_v < 0) {
123  error << "Illegal number of vehicles found vehicle";
124  log << vehicle.cant_v << "< 0 on vehicle " << vehicle.id;
125  continue;
126  }
127 
128  auto starting_site = Vehicle_node(
129  {node_id++, vehicle, Tw_node::NodeType::kStart});
130  auto ending_site = Vehicle_node(
131  {node_id++, vehicle, Tw_node::NodeType::kEnd});
132 
133  if (!(starting_site.is_start()
134  && ending_site.is_end())) {
135  error << "Illegal values found on vehcile";
136  return false;
137  }
138 
139  problem->add_node(starting_site);
140  problem->add_node(ending_site);
141 
142  for (int i = 0; i < vehicle.cant_v; ++i) {
143  m_trucks.push_back(Vehicle_pickDeliver(
144  m_trucks.size(),
145  vehicle.id,
146  starting_site,
147  ending_site,
148  vehicle.capacity,
149  vehicle.speed
150  ));
151  log << "inserting " << m_trucks.back().id();
152  pgassert((m_trucks.back().id() + 1) == m_trucks.size());
153  }
154  }
155  Identifiers<size_t> unused(m_trucks.size());
156  un_used = unused;
157  return true;
158 }
