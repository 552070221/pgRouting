70 void
71 Vehicle_node::evaluate(
72  const Vehicle_node &pred,
73  double cargoLimit,
74  double speed) {
76  /* time */
77 
78  m_travel_time = pred.travel_time_to(*this, speed);
79  m_arrival_time = pred.departure_time() + travel_time();
80  m_wait_time = is_early_arrival(arrival_time()) ?
81  opens() - m_arrival_time :
82  0;
83  m_departure_time = arrival_time() + wait_time() + service_time();
84 
85  /* time aggregates */
86  m_tot_travel_time = pred.total_travel_time() + travel_time();
87  m_tot_wait_time = pred.total_wait_time() + wait_time();
88  m_tot_service_time = pred.total_service_time() + service_time();
89 
90  /* cargo aggregates */
91  if (is_dump() && pred.cargo() >= 0) {
92  m_demand = -pred.cargo();
93  }
94  m_cargo = pred.cargo() + demand();
95 
96  /* cargo aggregates */
97 
98  m_twvTot = has_twv() ? pred.twvTot() + 1 : pred.twvTot();
99  m_cvTot = has_cv(cargoLimit) ? pred.cvTot() + 1 : pred.cvTot();
100  m_delta_time = departure_time() - pred.departure_time();
101 }
