36 void fetch_edge(
37  HeapTuple *tuple,
38  TupleDesc *tupdesc,
39  Column_info_t info[5],
40  int64_t *default_id,
41  float8 default_rcost,
42  pgr_edge_t *edge,
43  size_t *valid_edges) {
44  if (column_found(info[0].colNumber)) {
45  edge->id = pgr_SPI_getBigInt(tuple, tupdesc, info[0]);

47  edge->id = *default_id;
48  ++(*default_id);
49  }
50 
51  edge->source = pgr_SPI_getBigInt(tuple, tupdesc, info[1]);
52  edge->target = pgr_SPI_getBigInt(tuple, tupdesc, info[2]);
53  edge->cost = pgr_SPI_getFloat8(tuple, tupdesc, info[3]);
54 
55  if (column_found(info[4].colNumber)) {
56  edge->reverse_cost = pgr_SPI_getFloat8(tuple, tupdesc, info[4]);
57  } else {
58  edge->reverse_cost = default_rcost;
59  }
60 
61  *valid_edges = edge->cost < 0? *valid_edges: *valid_edges + 1;
62  *valid_edges = edge->reverse_cost < 0? *valid_edges: *valid_edges + 1;
63 }

108 get_edges_9_columns(
109  char *sql,
110  Pgr_edge_xy_t **edges,
111  size_t *total_edges,
112  bool normal) {
113  clock_t start_t = clock();
114 
115  const int tuple_limit = 1000000;
116 
117  size_t ntuples;
118  size_t total_tuples;
119  size_t valid_edges;
120 
121  Column_info_t info[9];
122 
123  info[0].name = strdup("id");
124  info[1].name = strdup("source");
125  info[2].name = strdup("target");
126  info[3].name = strdup("cost");
127  info[4].name = strdup("reverse_cost");
128  info[5].name = strdup("x1");
129  info[6].name = strdup("y1");
130  info[7].name = strdup("x2");
131  info[8].name = strdup("y2");
132 
133  int i;
134  for (i = 0; i < 3; ++i) {
135  info[i].colNumber = -1;
136  info[i].type = 0;
137  info[i].strict = true;
138  info[i].eType = ANY_INTEGER;
139  }
140  for (i = 3; i < 9; ++i) {
141  info[i].colNumber = -1;
142  info[i].type = 0;
143  info[i].strict = true;
144  info[i].eType = ANY_NUMERICAL;
145  }
146  /*
147  * reverse_cost is optional
148  */
149  info[4].strict = false;
150 
151 
152  void *SPIplan;
153  SPIplan = pgr_SPI_prepare(sql);
154 
155  Portal SPIportal;
156  SPIportal = pgr_SPI_cursor_open(SPIplan);
157 
158 
159  bool moredata = TRUE;
160  (*total_edges) = total_tuples = valid_edges = 0;
161 
162 
163  int64_t default_id = 0;
164  while (moredata == TRUE) {
165  SPI_cursor_fetch(SPIportal, TRUE, tuple_limit);
166  if (total_tuples == 0)
167  pgr_fetch_column_info(info, 9);
168 
169  ntuples = SPI_processed;
170  total_tuples += ntuples;
171 
172  if (ntuples > 0) {
173  if ((*edges) == NULL)
174  (*edges) = (Pgr_edge_xy_t *)palloc0(total_tuples * sizeof(Pgr_edge_xy_t));
175  else
176  (*edges) = (Pgr_edge_xy_t *)repalloc((*edges), total_tuples * sizeof(Pgr_edge_xy_t));
177 
178  if ((*edges) == NULL) {
179  elog(ERROR, "Out of memory");
180  }
181 
182  size_t t;
183  SPITupleTable *tuptable = SPI_tuptable;
184  TupleDesc tupdesc = SPI_tuptable->tupdesc;
185  PGR_DBG("processing %d edge tupĺes", ntuples);
186 
187  for (t = 0; t < ntuples; t++) {
188  HeapTuple tuple = tuptable->vals[t];
189  fetch_edge_with_xy(&tuple, &tupdesc, info,
190  &default_id, -1,
191  &(*edges)[total_tuples - ntuples + t],
192  &valid_edges, normal);
193  }
194  SPI_freetuptable(tuptable);
195  } else {
196  moredata = FALSE;
197  }
198  }
199 
200 
201  if (total_tuples == 0 || valid_edges == 0) {
202  PGR_DBG("NO edges found");
203  return;
204  }
205 
206  (*total_edges) = total_tuples;
207  PGR_DBG("Finish reading %ld edges, %ld", total_tuples, (*total_edges));
208  time_msg(" reading Edges with xy", start_t, clock());
209 }
